JavaScript: creates a duplicate-free version of an Array

July 8, 2022 · 86 words · unnamed | Suggest Changes

Case 1#

const array = [1, 2, 2, 3, 3, 3];

// https://developer.mozilla.org/en-US/docs/Web/JavaScript/Reference/Global_Objects/Set
const uniqueArray = [...new Set(array)];
console.log(uniqueArray); // [1, 2, 3]

Case 2#

const Champions = [
  {
    name: "Yasuo",
    age: 18,
  },
  {
    name: "Zed",
    age: 19,
  },
  {
    name: "Darius",
    age: 20,
  },
  {
    name: "Yasuo",
    age: 21,
  },
  {
    name: "Kha'Zix",
    age: 21,
  },
];

// https://developer.mozilla.org/en-US/docs/Web/JavaScript/Reference/Global_Objects/Array/map
const uniqueChampionsName = [
  ...new Set(Champions.map((element) => element.name)),
];
const uniqueChampionsAge = [
  ...new Set(Champions.map((element) => element.age)),
];

console.log(uniqueChampionsName); // ['Yasuo', 'Zed', 'Darius', "Kha'Zix"]
